115 double amon::Graph::globalClusteringCoefficient() {
116  std::vector < std::set<int> > L;
117  std::vector < std::pair<int, int> > nodes;
118 
119  for (int i = 0; i < (int)adj.size(); ++i) {
120  if (!validNodes[i]) continue;
121  nodes.push_back( std::make_pair(i, L.size()) );
122  L.push_back(std::set<int>());
123  for (auto e : adj[i]) {
124  L[i].insert(e.first);
125  }
126  }
127 
128  double triangles = 0.0, triplets = 0.0;
129  for (int i = 0; i < (int)nodes.size(); ++i) {
130  for (int j = i+1; j < (int)nodes.size(); ++j) {
131  int ii = nodes[i].second;
132  int jj = nodes[j].second;
133  // They do not connect to each other
134  if (!L[ii].count(jj) or !L[jj].count(ii)) continue;
135  // Size of union -> triplets
136  int sz = L[ii].size() + L[jj].size();
137  std::vector <int> I(sz), U(sz);
138  int usz = std::set_union(L[ii].begin(), L[ii].end(),
139  L[jj].begin(), L[jj].end(), U.begin()) - U.begin();
140  int isz = std::set_intersection(L[ii].begin(), L[ii].end(),
141  L[jj].begin(), L[jj].end(), I.begin()) - I.begin();
142  triangles += isz;
143  triplets += usz - 2;
144  }
145  }
146  triangles /= 3.0;
147  triplets -= triangles*3.0;
148  triplets /= 2.0;
149  triplets += triangles;
150  return triangles/triplets;
151 }
