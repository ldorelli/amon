138 double amon::Graph::globalClusteringCoefficient() {
139  std::vector < std::set<int> > L;
140  std::vector < std::pair<int, int> > nodes;
141 
142  for (int i = 0; i < (int)adj.size(); ++i) {
143  if (!validNodes[i]) continue;
144  nodes.push_back( std::make_pair(i, L.size()) );
145  L.push_back(std::set<int>());
146  for (auto e : adj[i]) {
147  L[i].insert(e.first);
148  }
149  }
150 
151  double triangles = 0.0, triplets = 0.0;
152  for (int i = 0; i < (int)nodes.size(); ++i) {
153  for (int j = i+1; j < (int)nodes.size(); ++j) {
154  int ii = nodes[i].second;
155  int jj = nodes[j].second;
156  // They do not connect to each other
157  if (!L[ii].count(jj) or !L[jj].count(ii)) continue;
158  // Size of union -> triplets
159  int sz = L[ii].size() + L[jj].size();
160  std::vector <int> I(sz), U(sz);
161  int usz = std::set_union(L[ii].begin(), L[ii].end(),
162  L[jj].begin(), L[jj].end(), U.begin()) - U.begin();
163  int isz = std::set_intersection(L[ii].begin(), L[ii].end(),
164  L[jj].begin(), L[jj].end(), I.begin()) - I.begin();
165  triangles += isz;
166  triplets += usz - 2;
167  }
168  }
169  triangles /= 3.0;
170  triplets -= triangles*3.0;
171  triplets /= 2.0;
172  triplets += triangles;
173  return triangles/triplets;
174 }

307 void amon::Graph::betweennessUnweighted(int x, std::vector<double>& res) {
308  std::vector <double> sigma(nodesCount, 0.0), dist(nodesCount, -1);
309  std::stack <int> S;
310  std::vector < std::vector<int> > P (nodesCount);
311  std::queue <int> Q;
312 
313  Q.push(x);
314  sigma[x] = 1.0;
315  dist[x] = 0;
316  while (!Q.empty()) {
317  int v = Q.front(); Q.pop();
318  S.push(v);
319  for (auto p : adj[v]) {
320  int w = p.first;
321  if (dist[w] == -1) {
322  dist[w] = dist[v] + 1;
323  Q.push(w);
324  }
325  if (dist[w] == dist[v] + 1) {
326  sigma[w] += sigma[v];
327  P[w].push_back(v);
328  }
329  }
330  }
331  std::vector <double> delta(nodesCount);
332  while (!S.empty()) {
333  int w = S.top();
334  for (int i = 0; i < (int)P[w].size(); ++i) {
335  int v = P[w][i];
336  delta[v] += (sigma[v]/sigma[w]) * (1 + delta[w]);
337  }
338  if (w != x) res[w] += delta[w];
339  S.pop();
340  }
341 }
